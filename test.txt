      document.getElementById("consumerDataOnly").innerHTML = `<div class="data-item"><span>Voltage</span><span class="data-value">${volt} V</span></div><div class="data-item"><span>Current</span><span class="data-value">${curr} A</span></div><div class="data-item"><span>Power</span><span class="data-value">${power} W</span></div><div class="data-item"><span>Energy</span><span class="data-value">${energy} kWh</span></div><div class="data-item"><span>PF</span><span class="data-value">${pf}</span></div><div class="data-item"><span>Freq</span><span class="data-value">${freq} Hz</span></div>`;
      if (!isNaN(powerNum)) addDataToChart(consumerOnlyChart, powerNum, timeLabel);
    }
    // Fault detection logic uses latest power values from charts or direct
    checkTheftCondition(powerNum);
  }

  let lastTransformerPower = 0;
  let lastConsumerPower = 0;

  function checkTheftCondition(consumerPowerW) {
    if (!currentTransformer) return;
    const transPowerRaw = safeVal(currentTransformer, 'Power', 0);
    const tPower = parseFloat(transPowerRaw);
    const cPower = parseFloat(consumerPowerW);
    if (isNaN(tPower) || isNaN(cPower)) return;
    lastTransformerPower = tPower;
    lastConsumerPower = cPower;
    const diff = tPower - cPower;
    if (diff > 30) {  // theft threshold >30W
      if (!faultTimer) {
        faultTimer = setTimeout(() => {
          logFaultEvent(tPower, cPower, diff);
          showAlert(`⚠️ POWER THEFT DETECTED! Loss of ${diff.toFixed(1)}W`, "danger");
          playAlertBeep();
          document.body.classList.add("blink-bg");
          faultActiveFlag = true;
          setTimeout(() => { document.body.classList.remove("blink-bg"); faultActiveFlag = false; }, 10000);
        }, 5000);
      }
    } else {
      if (faultTimer) { clearTimeout(faultTimer); faultTimer = null; }
      if (faultActiveFlag) { document.body.classList.remove("blink-bg"); faultActiveFlag = false; }
    }
  }

  function logFaultEvent(tp, cp, diff) {
    const tsStr = new Date().toLocaleString();
    faultLogArray.push({ Timestamp: tsStr, Transformer: tp, Consumer: cp, Difference: diff });
    storeFaultToDB(tp, cp, diff);
    const row = `<tr><td>${tsStr}</td><td>${tp.toFixed(1)}</td><td>${cp.toFixed(1)}</td><td style="color:var(--danger); font-weight:bold;">${diff.toFixed(1)}</td></tr>`;
    if (userType === "admin") document.querySelector("#faultTable tbody").innerHTML += row;
    if (userType === "consumer") document.querySelector("#consumerFaultTable tbody").innerHTML += row;
  }

  // Data listeners from firebase
  function loadDataRealtime() {
    const transRef = db.ref("ESP32_1");
    const consRef = db.ref("ESP32_2");

    transRef.on('value', snap => {
      if (snap.exists()) { currentTransformer = snap.val(); renderTransformer(currentTransformer); }
    });
    consRef.on('value', snap => {
      if (snap.exists()) { currentConsumer = snap.val(); renderConsumer(currentConsumer); }
    });

    // Periodic store to history
    if (dataStoreInterval) clearInterval(dataStoreInterval);
    dataStoreInterval = setInterval(() => {
      if (currentTransformer) storeDataPoint("transformer", currentTransformer);
      if (currentConsumer) storeDataPoint("consumer", currentConsumer);
    }, 35000);
  }

  function loadHistoryFor(device, tbodyId) {
    const ref = db.ref(`power_data/${device}`);
    ref.orderByKey().limitToLast(50).once('value').then(snap => {
      const tbody = document.getElementById(tbodyId);
      if (!tbody) return;
      tbody.innerHTML = "";
      if (!snap.exists()) { tbody.innerHTML = "<tr><td colspan='7'>No historical records</td></tr>"; return; }
      const items = [];
      snap.forEach(child => { items.push({ key: child.key, ...child.val() }); });
      items.sort((a,b) => b.key - a.key);
      items.forEach(rec => {
        const row = `<tr>
          <td>${new Date(parseInt(rec.key)).toLocaleString()}</td>
          <td>${safeVal(rec, 'Voltage', '-')}</td>
          <td>${safeVal(rec, 'Current', '-')}</td>
          <td>${safeVal(rec, 'Power', '-')}</td>
          <td>${safeVal(rec, 'Energy', '-')}</td>
          <td>${safeVal(rec, 'PF', safeVal(rec, 'PowerFactor', '-'))}</td>
          <td>${safeVal(rec, 'Freq', safeVal(rec, 'Frequency', '-'))}</td>
        </tr>`;
        tbody.innerHTML += row;
      });
    }).catch(() => {});
  }

  function loadFaultHistory() {
    const ref = db.ref('fault_logs');
    ref.orderByKey().limitToLast(50).once('value').then(snap => {
      const tbody = document.getElementById('faultHistoricalBody');
      if (!tbody) return;
      tbody.innerHTML = "";
      if (!snap.exists()) { tbody.innerHTML = "<tr><td colspan='4'>No fault logs</td></tr>"; return; }
      const logs = [];
      snap.forEach(child => { logs.push({ key: child.key, ...child.val() }); });
      logs.sort((a,b) => b.key - a.key);
      logs.forEach(log => {
        tbody.innerHTML += `<tr><td>${new Date(parseInt(log.key)).toLocaleString()}</td><td>${safeVal(log,'transformerPower','-')}</td><td>${safeVal(log,'consumerPower','-')}</td><td>${safeVal(log,'difference','-')}</td></tr>`;
      });
    });
  }

  window.downloadLog = function() {
    if (faultLogArray.length === 0) { showAlert("No fault events captured yet", "warning"); return; }
    let csv = "Timestamp,Transformer_Power_W,Consumer_Power_W,Difference_W\n";
    faultLogArray.forEach(r => { csv += `"${r.Timestamp}",${r.Transformer},${r.Consumer},${r.Difference}\n`; });
    const blob = new Blob([csv], { type: "text/csv" });
    const a = document.createElement("a");
    a.href = URL.createObjectURL(blob);
    a.download = `theft_log_${Date.now()}.csv`;
    a.click();
    URL.revokeObjectURL(a.href);
    showAlert("CSV exported", "success");
  };
